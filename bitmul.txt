public class bitmanipulation {
    public static void main(String[] args) {
        int n=4;
        int pos=2;
        int bitmask=1<<pos;
        if((bitmask & n)==0)
        {
             System.out.print("bit is zero");
        }
        else{
            System.out.print("bit is one");
        }


    }
    
}
